Inventory Schema
Schema::create('inventory', function (Blueprint $table) {
    $table->id();
    $table->foreignId('smartphone_id')->constrained()->onDelete('cascade');
    $table->date('purchase_date')->nullable();
    $table->string('supplier')->nullable();
    $table->decimal('purchase_unit_price', 20, 2)->nullable();
    $table->decimal('additional_costs', 20, 2)->nullable();
    $table->decimal('total_purchase_amount', 20, 2)->nullable();
    $table->string('storage_location')->nullable();
    $table->enum('status', ['in_stock', 'on_hold', 'sold', 'returned'])->default('in_stock');
    $table->date('return_date')->nullable();
    $table->timestamps();
});
